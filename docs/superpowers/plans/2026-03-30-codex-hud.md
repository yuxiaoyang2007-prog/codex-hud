# Codex HUD Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local Codex HUD that adds a terminal footer for Codex CLI sessions and a VS Code companion surface powered by one shared telemetry core.

**Architecture:** Use a TypeScript workspace with three deliverables: a shared `core` package for events and snapshots, a `terminal` wrapper that launches `codex` in a PTY and renders a footer HUD, and a `vscode` extension that watches the same snapshot file. Add a local Codex plugin with hooks for precise lifecycle events, then layer fallback session-log parsing on top so the HUD still works when hook coverage is incomplete.

**Tech Stack:** Node.js 22+, TypeScript, npm workspaces, Vitest, node-pty, VS Code Extension API, local Codex plugin hooks, JSON snapshot files, Unix domain sockets on macOS/Linux

- [ ] **Step 2: Add the event union and reducer implementation**

```ts
export type HudEvent =
  | { type: 'session.start'; at: string; model?: string; reasoningEffort?: string }
  | { type: 'phase.update'; at: string; phase: HudPhase }
  | { type: 'tool.start'; at: string; toolName: string }
  | { type: 'tool.finish'; at: string; toolName: string; success: boolean }
  | { type: 'plan.update'; at: string; currentStep: string | null; completedSteps: number; totalSteps: number }
  | { type: 'subagent.update'; at: string; active: number; lastEvent: string }
  | { type: 'warning'; at: string; message: string };
```

```ts
import type { HudEvent, HudSnapshot } from './schema';

export function applyHudEvent(snapshot: HudSnapshot, event: HudEvent): HudSnapshot {
  const next = {
    ...snapshot,
    session: {
      ...snapshot.session,
      lastUpdatedAt: event.at
    }
  };

  switch (event.type) {
    case 'session.start':
      next.session.startedAt = event.at;
      next.session.model = event.model ?? null;
      next.session.reasoningEffort = event.reasoningEffort ?? null;
      return next;
    case 'tool.start':
      next.status.phase = 'tool-running';
      next.tool.activeName = event.toolName;
      next.tool.startedAt = event.at;
      next.tool.elapsedMs = 0;
      return next;
    case 'tool.finish':
      next.status.phase = event.success ? 'idle' : 'error';
      next.tool.elapsedMs = snapshot.tool.startedAt
        ? Date.parse(event.at) - Date.parse(snapshot.tool.startedAt)
        : 0;
      next.tool.activeName = null;
      next.tool.startedAt = null;
      return next;
    case 'plan.update':
      next.plan = {
        currentStep: event.currentStep,
        completedSteps: event.completedSteps,
        totalSteps: event.totalSteps
      };
      return next;
    case 'subagent.update':
      next.subagents = {
        active: event.active,
        lastEvent: event.lastEvent
      };
      return next;
    case 'warning':
      next.warnings = [...snapshot.warnings, event.message].slice(-5);
      return next;
    case 'phase.update':
      next.status.phase = event.phase;
      return next;
  }
}
```

```ts
export async function writeSnapshot(snapshotPath: string, snapshot: HudSnapshot): Promise<void> {
  await mkdir(dirname(snapshotPath), { recursive: true });
```

```ts
import net from 'node:net';
import { createEmptySnapshot } from './schema';
import { applyHudEvent } from './reducer';
import { writeSnapshot } from './state-store';

export function createHudSocketServer(socketPath: string, snapshotPath: string) {
  let snapshot = createEmptySnapshot('pending-session');

  return net.createServer((connection) => {
    connection.setEncoding('utf8');
    connection.on('data', async (chunk) => {
      for (const line of chunk.split('\n').filter(Boolean)) {
        const event = JSON.parse(line);
        if (snapshot.session.id === 'pending-session' && event.sessionId) {
          snapshot = { ...snapshot, session: { ...snapshot.session, id: event.sessionId } };
        }
        snapshot = applyHudEvent(snapshot, event);
        await writeSnapshot(snapshotPath, snapshot);
      }
    });
  });
}
```

```js
export function buildHookEvent(kind, env) {
  const at = env.CODEX_EVENT_AT ?? new Date().toISOString();
  const sessionId = env.CODEX_HUD_SESSION_ID;

  if (kind === 'pre-tool-use') {
    return {
      sessionId,
      type: 'tool.start',
      toolName: env.CODEX_TOOL_NAME,
      at
    };
  }

  if (kind === 'post-tool-use') {
    return {
      sessionId,
      type: 'tool.finish',
      toolName: env.CODEX_TOOL_NAME,
      success: env.CODEX_TOOL_EXIT_CODE === '0',
      at
    };
  }

  return {
    sessionId,
    type: kind === 'session-start' ? 'session.start' : 'phase.update',
    phase: kind === 'user-prompt-submit' ? 'thinking' : 'idle',
    at
  };
}
```

```ts
  const line1 = [
    `session:${snapshot.session.id}`,
    `model:${snapshot.session.model ?? 'n/a'}`,

    `tool:${snapshot.tool.activeName ?? 'none'}`

    `plan:${snapshot.plan.completedSteps}/${snapshot.plan.totalSteps}`,
    `step:${snapshot.plan.currentStep ?? 'n/a'}`,
    `agents:${snapshot.subagents.active}`,
    `warn:${snapshot.warnings.at(-1) ?? 'none'}`
  ].join(' | ');

  return [line1.slice(0, columns), line2.slice(0, columns)].join('\n');
}
```

```ts
import type { HudEvent } from './schema';

export function parseCodexSessionLines(lines: string[]): HudEvent[] {
  const events: HudEvent[] = [];
  let activeAgents = 0;

  for (const line of lines) {
    const record = JSON.parse(line);
    const name = record?.payload?.name;
    const args = record?.payload?.arguments;
    const at = record?.timestamp ?? new Date().toISOString();

    if (name === 'update_plan' && Array.isArray(args?.plan)) {
      const current = args.plan.find((item: { status: string }) => item.status === 'in_progress') ?? null;
      const completed = args.plan.filter((item: { status: string }) => item.status === 'completed').length;

      events.push({
        type: 'plan.update',
        at,
        currentStep: current?.step ?? null,
        completedSteps: completed,
        totalSteps: args.plan.length
      });
    }

    if (name === 'spawn_agent') {
      activeAgents += 1;
      events.push({ type: 'subagent.update', at, active: activeAgents, lastEvent: 'spawn_agent' });
    }

    if (name === 'close_agent' && activeAgents > 0) {
      activeAgents -= 1;
      events.push({ type: 'subagent.update', at, active: activeAgents, lastEvent: 'close_agent' });
    }
  }

  return events;
}
```

```ts
import * as vscode from 'vscode';
import { readFile } from 'node:fs/promises';

export async function readSnapshot(snapshotPath: string) {
  return JSON.parse(await readFile(snapshotPath, 'utf8'));
}

export function formatStatusBarText(snapshot: {
  session: { model: string | null };
  status: { phase: string };
  tool: { activeName: string | null };
  plan: { completedSteps: number; totalSteps: number };
}) {
  return [
    `Codex HUD`,
    snapshot.session.model ?? 'n/a',
    snapshot.status.phase,
    snapshot.tool.activeName ?? 'no-tool',
    `${snapshot.plan.completedSteps}/${snapshot.plan.totalSteps}`
  ].join(' · ');
}
```

```ts
export async function activate(context: vscode.ExtensionContext) {
  const item = vscode.window.createStatusBarItem(vscode.StatusBarAlignment.Left, 50);
  item.command = 'codexHud.showDetails';
  item.show();

  const snapshotPath = vscode.workspace.getConfiguration('codexHud').get<string>('snapshotPath');
  if (!snapshotPath) {
    item.text = 'Codex HUD · no snapshot';
    return;
  }

  const snapshot = await readSnapshot(snapshotPath);
  item.text = formatStatusBarText(snapshot);
  context.subscriptions.push(item);
}
```
